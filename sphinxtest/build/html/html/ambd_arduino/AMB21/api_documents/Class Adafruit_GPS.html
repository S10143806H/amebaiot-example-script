Example Code

This example code from Adafruit demonstrates GPS modules using MTK3329/MTK3339 driver. This code shows how to listen to the GPS module in an interrupt which allows the program to have more ‘freedom’ – just parse when a new NMEA sentence is available! Then access data when desired.

Adafruit_GPS_parsing.ino
  1 #include <Adafruit_GPS.h>
  2 #include <SoftwareSerial.h>
  3
  4 // If you're using a GPS module:
  5 // Connect the GPS Power pin to 3.3V
  6 // Connect the GPS Ground pin to ground
  7 // Connect the GPS TX (transmit) pin to Digital 0
  8 // Connect the GPS RX (receive) pin to Digital 1
  9 #if defined(BOARD_RTL8195A)
 10 SoftwareSerial mySerial(0, 1);
 11 #elif defined(BOARD_RTL8710)
 12 SoftwareSerial mySerial(17, 5); // RTL8710 need change GPS TX/RX to pin 17 and 5
 13 #else
 14 SoftwareSerial mySerial(0, 1);
 15 #endif
 16
 17 Adafruit_GPS GPS(&mySerial);
 18
 19 // Set GPSECHO to 'false' to turn off echoing the GPS data to the Serial console
 20 // Set to 'true' if you want to debug and listen to the raw GPS sentences.
 21 #define GPSECHO  false
 22
 23 void setup()
 24 {
 25 Serial.begin(38400);
 26 Serial.println("Adafruit GPS library basic test!");
 27
 28 // 9600 NMEA is the default baud rate for Adafruit MTK GPS's- some use 4800
 29 GPS.begin(9600);
 30
 31 // uncomment this line to turn on RMC (recommended minimum) and GGA (fix data) including altitude
 32 GPS.sendCommand(PMTK_SET_NMEA_OUTPUT_RMCGGA);
 33 // uncomment this line to turn on only the "minimum recommended" data
 34 //GPS.sendCommand(PMTK_SET_NMEA_OUTPUT_RMCONLY);
 35 // For parsing data, we don't suggest using anything but either RMC only or RMC+GGA since
 36 // the parser doesn't care about other sentences at this time
 37
 38 // Set the update rate
 39 GPS.sendCommand(PMTK_SET_NMEA_UPDATE_1HZ);   // 1 Hz update rate
 40 // For the parsing code to work nicely and have time to sort thru the data, and
 41 // print it out we don't suggest using anything higher than 1 Hz
 42
 43 // Request updates on antenna status, comment out to keep quiet
 44 GPS.sendCommand(PGCMD_ANTENNA);
 45
 46 delay(1000);
 47 // Ask for firmware version
 48 mySerial.println(PMTK_Q_RELEASE);
 49 }
 50
 51 uint32_t timer = millis();
 52 void loop()                     // run over and over again
 53 {
 54 // in case you are not using the interrupt above, you'll
 55 // need to 'hand query' the GPS, not suggested :(
 56 // read data from the GPS in the 'main loop'
 57 char c = GPS.read();
 58 // if you want to debug, this is a good time to do it!
 59 if (GPSECHO)
 60     if (c) Serial.print(c);
 61
 62 // if a sentence is received, we can check the checksum, parse it...
 63 if (GPS.newNMEAreceived()) {
 64     // a tricky thing here is if we print the NMEA sentence, or data
 65     // we end up not listening and catching other sentences!
 66     // so be very wary if using OUTPUT_ALLDATA and trytng to print out data
 67     //Serial.println(GPS.lastNMEA());   // this also sets the newNMEAreceived() flag to false
 68
 69     if (!GPS.parse(GPS.lastNMEA()))   // this also sets the newNMEAreceived() flag to false
 70     return;  // we can fail to parse a sentence in which case we should just wait for another
 71 }
 72
 73 // if millis() or timer wraps around, we'll just reset it
 74 if (timer > millis())  timer = millis();
 75
 76 // approximately every 2 seconds or so, print out the current stats
 77 if (millis() - timer > 2000) {
 78     timer = millis(); // reset the timer
 79
 80     Serial.print("\nTime: ");
 81     Serial.print(GPS.hour, DEC); Serial.print(':');
 82     Serial.print(GPS.minute, DEC); Serial.print(':');
 83     Serial.print(GPS.seconds, DEC); Serial.print('.');
 84     Serial.println(GPS.milliseconds);
 85     Serial.print("Date: ");
 86     Serial.print(GPS.day, DEC); Serial.print('/');
 87     Serial.print(GPS.month, DEC); Serial.print("/20");
 88     Serial.println(GPS.year, DEC);
 89     Serial.print("Fix: "); Serial.print((int)GPS.fix);
 90     Serial.print(" quality: "); Serial.println((int)GPS.fixquality);
 91     if (GPS.fix) {
 92     Serial.print("Location: ");
 93     Serial.print(GPS.latitude, 4); Serial.print(GPS.lat);
 94     Serial.print(", ");
 95     Serial.print(GPS.longitude, 4); Serial.println(GPS.lon);
 96     Serial.print("Location (in degrees, works with Google Maps): ");
 97     Serial.print(GPS.latitudeDegrees, 4);
 98     Serial.print(", ");
 99     Serial.println(GPS.longitudeDegrees, 4);
100
101     Serial.print("Speed (knots): "); Serial.println(GPS.speed);
102     Serial.print("Angle: "); Serial.println(GPS.angle);
103     Serial.print("Altitude: "); Serial.println(GPS.altitude);
104     Serial.print("Satellites: "); Serial.println((int)GPS.satellites);
105     }
106 }
107 }

SoftSerial is using hardware serial so pin mapping cannot be altered.
